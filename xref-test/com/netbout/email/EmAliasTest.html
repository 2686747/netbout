29  import com.jcabi.email.Envelope;
30  import com.jcabi.email.Postman;
31  import com.netbout.mock.MkBase;
32  import com.netbout.spi.Alias;
33  import com.netbout.spi.Bout;
34  import java.io.ByteArrayOutputStream;
35  import javax.mail.Message;
36  import javax.mail.internet.MimeMultipart;
37  import org.hamcrest.MatcherAssert;
38  import org.hamcrest.Matchers;
39  import org.junit.Test;
40  import org.mockito.ArgumentCaptor;
41  import org.mockito.Mockito;
42  
43  /**
44   * Test case for {@link EmAlias}.
45   * @author Yegor Bugayenko (yegor@tpc2.com)
46   * @version $Id$
47   * @since 2.12.2
48   */
49  public final class EmAliasTest {
50  
51      /**
52       * EmAlias can send an email.
53       * @throws Exception If there is some problem inside
54       */
55      @Test
56      public void sendsEmail() throws Exception {
57          final Postman postman = Mockito.mock(Postman.class);
58          final MkBase base = new MkBase();
59          final Alias alias = new EmAlias(base.randomAlias(), postman);
60          final Bout bout = alias.inbox().bout(alias.inbox().start());
61          bout.friends().invite(base.randomAlias().name());
62          bout.messages().post("how are you?");
63          final ArgumentCaptor<Envelope> captor =
64              ArgumentCaptor.forClass(Envelope.class);
65          Mockito.verify(postman).send(captor.capture());
66          final Message msg = captor.getValue().unwrap();
67          MatcherAssert.assertThat(msg.getFrom().length, Matchers.is(1));
68          MatcherAssert.assertThat(msg.getAllRecipients().length, Matchers.is(1));
69          MatcherAssert.assertThat(msg.getSubject(), Matchers.startsWith("#1: "));
70          final ByteArrayOutputStream baos = new ByteArrayOutputStream();
71          MimeMultipart.class.cast(msg.getContent()).writeTo(baos);
72          MatcherAssert.assertThat(
73              baos.toString(), Matchers.containsString("how are you")
74          );
75      }
